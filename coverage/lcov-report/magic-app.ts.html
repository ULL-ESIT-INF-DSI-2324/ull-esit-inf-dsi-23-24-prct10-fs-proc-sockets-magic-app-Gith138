    resultado += `Id: ${carta_json.id}\n`;

    resultado += `Linea de tipo: ${carta_json.tipo}\n`;
    resultado += `Rareza: ${carta_json.rareza}\n`;
    resultado += `Texto de reglas: ${carta_json.texto_reglas}\n`;
